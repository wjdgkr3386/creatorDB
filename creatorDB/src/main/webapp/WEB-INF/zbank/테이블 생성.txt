CREATE TABLE CREATOR (
  NICKNAME            VARCHAR2(50)       NOT NULL,
  CREATOR_IMAGE       VARCHAR2(4000)         NULL,
  DESCRIPTION         VARCHAR2(4000)         NULL,
  SUBSCRIBER_COUNT    VARCHAR2(4000)         NULL,
  VIDEO_COUNT         VARCHAR2(4000)         NULL,
  VIEW_COUNT          VARCHAR2(4000)         NULL,
  COUNTRY             VARCHAR2(4000)         NULL,
  CHANNEL_ID          VARCHAR2(4000)     NOT NULL,
  LAST_UPDATE         DATE              DEFAULT SYSDATE               NOT NULL
);

CREATE TABLE USERS (
  NAME             VARCHAR2(50)       NOT NULL,
  MID              VARCHAR2(50)           NULL,
  PWD              VARCHAR2(100)      NOT NULL,
  JUMIN_NUMBER1    VARCHAR2(6)        NOT NULL,
  JUMIN_NUMBER2    VARCHAR2(7)        NOT NULL,
  PHONE_NUMBER1    VARCHAR2(3)        NOT NULL,
  PHONE_NUMBER2    VARCHAR2(4)        NOT NULL,
  PHONE_NUMBER3    VARCHAR2(5)        NOT NULL,
  EMAIL            VARCHAR2(200)      NOT NULL,
  REGION           VARCHAR2(10)       NOT NULL,
  ADDRESS          VARCHAR2(1000)     NOT NULL,
  R_CODE           VARCHAR2(100)      NOT NULL
);
